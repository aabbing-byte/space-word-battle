            useEffect(() => {

                let interval;

                if (view === 'battle') {

                    if (phase === 'answering' && timeLeft > 0) {

                        interval = setInterval(() => setTimeLeft(prev => prev - 1), 1000);

                    } else if (phase === 'answering' && timeLeft === 0) {

                        setPhase('buffer');

                        setBufferLeft(3);

                    } else if (phase === 'buffer' && bufferLeft > 0) {

                        interval = setInterval(() => setBufferLeft(prev => prev - 1), 1000);

                    } else if (phase === 'buffer' && bufferLeft === 0) {

                        if (battleRound < totalRounds - 1) {

                            setBattleRound(prev => prev + 1);

                            setPhase('answering');

                            setTimeLeft(10);

                        } else {

                            setView('result');

                        }

                    }

                }

                return () => clearInterval(interval);

            }, [view, phase, timeLeft, bufferLeft, battleRound, totalRounds]);


            const getRandomOptions = useCallback((correctWord) => {

                const others = WORDS.filter(w => w.en !== correctWord.en);

                const shuffled = others.sort(() => Math.random() - 0.5).slice(0, 3);

                return [...shuffled, correctWord].sort(() => Math.random() - 0.5);

            }, []);


            useEffect(() => {

                if (view === 'battle') {

                    const currentWord = WORDS[battleRound % WORDS.length];

                    setBattleOptions(getRandomOptions(currentWord));

                    setBattleWinner(null);

                }

            }, [view, battleRound, getRandomOptions]);


            const startBattle = () => {

                setBattleRound(0);

                setBattleScores({ teamA: 0, teamB: 0 });

                setPhase('answering');

                setTimeLeft(10);

                setView('battle');

            };


            const handleBattleAnswer = (team, selectedEn) => {

                if (phase !== 'answering') return;

                const correctEn = WORDS[battleRound % WORDS.length].en;

                

                if (selectedEn === correctEn) {

                    setBattleWinner(team);

                    setBattleScores(prev => ({ ...prev, [team]: prev[team] + 1 }));

                    setPhase('buffer');

                    setBufferLeft(3);

                }

            };

                            <button key={num} onClick={() => setTotalRounds(num)} className={`px-6 py-2 rounded-xl font-black transition-all ${totalRounds === num ? 'bg-blue-600 text-white shadow-lg' : 'text-slate-400 hover:bg-slate-700'}`}>{num} 題</button>

                        ))}

                    </div>

                            <Icons.Zap className="mb-2 text-yellow-300 animate-pulse" size={32} />

                            <span className="font-bold">開始限時對戰</span>
